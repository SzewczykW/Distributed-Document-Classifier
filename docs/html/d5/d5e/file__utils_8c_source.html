84void broadcast_dictionary(char *keywords[], int num_keywords)
85{
86 gaspi_rank_t size;
87 GASPI_CHECK(gaspi_proc_num(&size));
88
89 gaspi_pointer_t dict_seg_ptr;
90 GASPI_CHECK(gaspi_segment_ptr(DICT_SEGMENT_ID, &dict_seg_ptr));
91
92 // Zapisz liczbę słów kluczowych
93 memcpy(dict_seg_ptr, &num_keywords, sizeof(int));
94
95 // Zapisz słowa kluczowe
96 for (int i = 0; i < num_keywords; i++)
97 {
98 strncpy((char *) dict_seg_ptr + sizeof(int) + i * MAX_WORD_LEN, keywords[i],
99 MAX_WORD_LEN - 1);
100 }
101
102 // Wyślij do wszystkich workerów
103 for (gaspi_rank_t rank = 1; rank < size; rank++)
104 {
105 GASPI_CHECK(gaspi_write(DICT_SEGMENT_ID, 0, rank, DICT_SEGMENT_ID, 0,
106 sizeof(int) + num_keywords * MAX_WORD_LEN, 0,
107 GASPI_BLOCK));
108 }
109
110 GASPI_CHECK(gaspi_wait(0, GASPI_BLOCK));
111}
